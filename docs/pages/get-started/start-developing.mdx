const prizes = [
  { name: "A 獎", chance: 1, remaining: 1 },
  { name: "B 獎", chance: 5, remaining: 3 },
  { name: "C 獎", chance: 20, remaining: 10 },
  { name: "D 獎", chance: 74, remaining: Infinity }
];

function drawPrize() {
  let random = Math.random() * 100;
  let cumulativeChance = 0;

  for (let prize of prizes) {
    cumulativeChance += prize.chance;
    if (random <= cumulativeChance && prize.remaining !== 0) {
      if (prize.remaining !== Infinity) prize.remaining--;
      return `恭喜你抽中了 ${prize.name}！`;
    }
  }
  return "很遺憾，沒有中獎！";
}

console.log(drawPrize());
